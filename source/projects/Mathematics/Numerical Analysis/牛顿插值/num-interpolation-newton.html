伪代码

输入x[i],y[i],i=0,1,...,n
// 求差商表，各阶差商直接存在y[i]单元内
for i=1 to n
   for j=n to i
       y[j]=(y[j]-y[j-1])/(x[j]-x[j-i]) // 得到的i阶差商，直接存在y[j]单元上
   end for
end for
// 求值，在x处
t=y[n]
for i=n-1 to 0
   t = t*(x-x[i])+y[i]
end for
// 现在t 就是N_n(x)
